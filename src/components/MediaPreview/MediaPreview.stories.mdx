import { Canvas, Meta, Story, ArgsTable } from '@storybook/addon-docs'
import { Button } from './../../';
import { MediaPreview } from './MediaPreview';

export const Template = (args) => {
	const [showPreview, setShowPreview] = React.useState(false);
	const [activeIndex, setActiveIndex] = React.useState(0);
	
	return(
		<div>
			<Button label="Show Preview" onClick={() => setShowPreview(true)} />
			<MediaPreview { ...args } show={showPreview} onClose={() => setShowPreview(false)} onChangeIndex={(i) => setActiveIndex(i)}/>
			current index: { activeIndex }
		</div>
	)
};

<Meta title='Components/MediaPreview' component={ MediaPreview } />
