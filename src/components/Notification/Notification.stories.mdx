import { Canvas, Meta, Story, ArgsTable } from '@storybook/addon-docs'
import { useNotifications } from './NotificationContext';
import { Notification } from './Notification';
import { Button, Text, TextNotification } from './../../';

export const Template = (args) => {
	const { addNotification } = useNotifications();

	const notificationList = [
		{
			message: (
				<TextNotification
					title="Ihr Profil wurde aktualisiert"
					body="Ihr Team hat Änderungen an Ihrem Unternehmensprofil vorgenommen, seit Sie sich zuletzt angemeldet haben."
				>
					<>
						<Button type="link" label="Später"/>
						<Button type="link" label="Änderungen ansehen"/>
					</>
				</TextNotification>
			),
			options: {}
		},
		{
			message: (

				<TextNotification
					title="Profil erfolgreich aktualisiert"
					body="Ihre Änderungen wurden gespeichert und Ihr Profil ist aktiv."
				>
					<>
						<Button type="link" label="Später"/>
						<Button type="link" label="Alle Änderungen ansehen"/>
					</>
				</TextNotification>
			),
			options: { type: "success" }
		},
		{
			message: (

				<TextNotification
					title="Es gibt ein neues Update!"
					body="Sehen Sie sich die neue Dashboard-Ansicht an. Seiten werden jetzt schneller geladen."
				>
					<>
						<Button type="link" label="Später"/>
						<Button type="link" label="Alle Änderungen ansehen"/>
					</>
				</TextNotification>
			),
			options: { type: "warning" }
		},
		{
			message: (
				<TextNotification
					title="Die Karte wurde nicht veröffentlicht"
					body="Durch das Entfernen aller Benutzer wurde dieses Projekt nicht veröffentlicht. Fügen Sie Benutzer hinzu, um es erneut zu veröffentlichen."
				>
					<>
						<Button type="link" label="Rückgängig machen"/>
					</>
				</TextNotification>
			),
			options: { type: "error" }
		}
	];

	

	const handleClick = () => {
		const randomMessage = notificationList[Math.floor(Math.random() * notificationList.length)];
		addNotification(randomMessage.message, randomMessage.options);
	};

	return (
		<Button onClick={handleClick} type="primary" label="Create Notification" />
	);
};
